#./configure \
    --prefix=/usr/local/apache \                            #定义一个目录,存放服务器上的文件,也就是apache的安装目录.默认使用/usr/local/apache.
    --enable-so \					    #参数是使httpd服务能够动态加载模块功能
    --enable-rewrite=shared \				    #是使httpd服务具有网页地址重写功能。或者不要后面的=shared写成--enable-rewrite
    --with-mpm=prefork \                                    #让apache以prefork方式运行,还有worker方式
    --with-apr=/usr/local/apr \                             #用于apr的configure脚本的选项：可选的额外程序包--with-apr=PATH指定apr的安装目录
    --with-apr-util=/usr/local/apr-util \  		    #用于apr-util的configure脚本的选项：可选的额外程序包--with-apr-util=PATH指定apr-util的安装目录
    --with-pcre=/usr/local/pcre 			    #用于pcre的configure脚本的选项：可选的额外程序包--with-pcre=PATH指定pcre的安装目录

常用编译参数一览
/configure   //配置源代码树
Cprefix=/usr/local/apache2   //体系无关文件的顶级安装目录PREFIX ，也就Apache的安装目录。
Cenable-module=so   //打开 so 模块，so 模块是用来提 DSO 支持的 apache 核心模块
Cenable-deflate=shared   //支持网页压缩
Cenable-expires=shared   //支持 HTTP 控制
Cenable-rewrite=shared   //支持 URL 重写
Cenable-cache //支持缓存
Cenable-file-cache //支持文件缓存
Cenable-mem-cache //支持记忆缓存
Cenable-disk-cache //支持磁盘缓存
Cenable-static-support   //支持静态连接(默认为动态连接)
Cenable-static-htpasswd   //使用静态连接编译 htpasswd C 管理用于基本认证的用户文件
Cenable-static-htdigest   //使用静态连接编译 htdigest C 管理用于摘要认证的用户文件
Cenable-static-rotatelogs   //使用静态连接编译 rotatelogs C 滚动 Apache 日志的管道日志程序
Cenable-static-logresolve   //使用静态连接编译 logresolve C 解析 Apache 日志中的IP地址为主机名
Cenable-static-htdbm   //使用静态连接编译 htdbm C 操作 DBM 密码数据库
Cenable-static-ab   //使用静态连接编译 ab C Apache HTTP 服务器性能测试工具
Cenable-static-checkgid   //使用静态连接编译 checkgid
Cdisable-cgid   //禁止用一个外部 CGI 守护进程执行CGI脚本
Cdisable-cgi   //禁止编译 CGI 版本的 PHP
Cdisable-userdir   //禁止用户从自己的主目录中提供页面
Cwith-mpm=worker // 让apache以worker方式运行
Cenable-authn-dbm=shared // 对动态数据库进行操作。Rewrite时需要。

以下是分门别类的更多参数注解，与上面的会有重复
用于apr的configure脚本的选项：
可选特性
--enable-experimental-libtool
启用试验性质的自定义libtool
--disable-libtool-lock
取消锁定(可能导致并行编译崩溃)
--enable-debug
启用调试编译，仅供开发人员使用。
--enable-maintainer-mode
打开调试和编译时警告，仅供开发人员使用。
--enable-profile
打开编译profiling(GCC)
--enable-pool-debug[=yes|no|verbose|verbose-alloc|lifetime|owner|all]
打开pools调试
--enable-malloc-debug
打开BeOS平台上的malloc_debug
--disable-lfs
在32-bit平台上禁用大文件支持(large file support)
--enable-nonportable-atomics
若只打算在486以上的CPU上运行Apache ，那么使用该选项可以启用更加高效的基于互斥执行的原子操作。
--enable-threads
启用线程支持，在线程型的MPM上必须打开它
--disable-threads
禁用线程支持，如果不使用线程化的MPM ，可以关闭它以减少系统开销。
--disable-dso
禁用DSO支持
--enable-other-child
启用可靠子进程支持
--disable-ipv6
禁用IPv6支持
可选的额外程序包
--with-gnu-ld
指定C编译器使用 GNU ld
--with-pic
只使用 PIC/non-PIC 对象[默认为两者都使用]
--with-tags[=TAGS]
包含额外的配置
--with-installbuilddir=DIR
指定APR编译文件的存放位置(默认值为：’${datadir}/build’)
--without-libtool
禁止使用libtool连接库文件
--with-efence[=DIR]
指定Electric Fence的安装目录
--with-sendfile
强制使用sendfile(译者注：Linux2.4/2.6内核都支持)
--with-egd[=DIR]
使用EDG兼容的socket
--with-devrandom[=DEV]
指定随机设备[默认为：/dev/random]
用于apr-util的configure脚本的选项：
可选的额外程序包
--with-apr=PATH
指定APR的安装目录(Cprefix选项值或apr-config的路径)
--with-ldap-include=PATH
ldap包含文件目录(带结尾斜线)
--with-ldap-lib=PATH
ldap库文件路径
--with-ldap=library
使用的ldap库
--with-dbm=DBM
选择使用的DBM类型DBM={sdbm,gdbm,ndbm,db,db1,db185,db2,db3,db4,db41,db42,db43,db44}
--with-gdbm=PATH
指定GDBM的位置
--with-ndbm=PATH
指定NDBM的位置
--with-berkeley-db=PATH
指定Berkeley DB的位置
--with-pgsql=PATH
指定PostgreSQL的位置
--with-mysql=PATH
参看INSTALL.MySQL文件的内容
--with-sqlite3=PATH
指定sqlite3的位置
--with-sqlite2=PATH
指定sqlite2的位置
--with-expat=PATH
指定Expat的位置或’builtin’
--with-iconv=PATH
iconv的安装目录
