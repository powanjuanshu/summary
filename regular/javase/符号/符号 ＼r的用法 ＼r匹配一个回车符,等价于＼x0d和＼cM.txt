package threedes;
public class Main4 {
	public static void main(String[] args) {
		boolean a1 = "\r".matches("\r"); // \r匹配一个回车符。等价于\x0d和\cM
		boolean a2 = "\n".matches("\r"); // \r匹配一个回车符。等价于\x0d和\cM
		System.out.println(a1); // true
		System.out.println(a2); // false
	}
}

package threedes;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
public class Main44 {
	public static void main(String[] args) {
		Pattern pattern = Pattern.compile("\r"); // \r匹配一个回车符。等价于\x0d和\cM
		Matcher m1 = pattern.matcher("\r");
		Matcher m2 = pattern.matcher("\n");
		boolean a = m1.matches(); // 不要写成m1.find，find是只要找到就是true，匹配一部分，matches是匹配所有
		boolean b = m2.matches(); // 不要写成m1.find，find是只要找到就是true，匹配一部分，matches是匹配所有
		System.out.println(a); // true
		System.out.println(b); // false
	}
}
